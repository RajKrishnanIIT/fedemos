  • Use conditional statements and loops effectively

Lesson 8.1: Conditional Statements

  • if/else statements
  • Switch statements
  • Ternary operator
  • Truthy and falsy values

Example

let age = 18;
if (age >= 18) {
  console.log('Adult');
} else {
  console.log('Minor');
}

let color = 'red';
switch (color) {
  case 'red':
    console.log('Color is red');
    break;
  case 'blue':
    console.log('Color is blue');
    break;
  default:
    console.log('Unknown color');
}

let isMember = true;
let fee = isMember ? '$5' : '$10';
console.log(fee);

Lesson 8.2: Loops

  • for loops (traditional and for...of)
  • while and do...while loops
  • Breaking and continuing loops
  • Nested loops

Example

for (let i = 0; i < 5; i++) {
  console.log(i);
}

const arr = [1, 2, 3];
for (const num of arr) {
  console.log(num);
}

let count = 0;
while (count < 3) {
  console.log(count);
  count++;
}

do {
  console.log(count);
  count--;
} while (count > 0);

Lesson 8.3: Arrays

  • Creating and accessing arrays
  • Array methods: push, pop, shift, unshift
  • Array iteration: forEach, map, filter, find
  • Multi-dimensional arrays

Example

let fruits = ['apple', 'banana'];
fruits.push('orange');
fruits.pop();
fruits.unshift('grape');
fruits.shift();

fruits.forEach(fruit => console.log(fruit));
let lengths = fruits.map(fruit => fruit.length);
let filtered = fruits.filter(fruit => fruit.startsWith('a'));
let found = fruits.find(fruit => fruit === 'banana');

let matrix = [[1,2],[3,4]];
console.log(matrix[0][1]);

Lesson 8.4: Objects

  • Object literals and property access
  • Methods and the 'this' keyword
  • Object constructors and prototypes (basic introduction)
  • JSON format

Example

let person = {
  name: 'Alice',
  greet: function() {
    return 'Hello, ' + this.name;
  }
};
console.log(person.greet());

function Car(make) {
  this.make = make;
}
Car.prototype.honk = function() {
  return 'Beep!';
};
let myCar = new Car('Toyota');
console.log(myCar.honk());

let json = JSON.stringify(person);
console.log(json);
